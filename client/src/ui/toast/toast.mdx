## Usage

```tsx
<>
	<ToastRegion>
		<ToastList />
	</ToastRegion>

	<Button
		variant="outline"
		onClick={() => {
			toaster.show(props => (
				<Toast toastId={props.toastId}>
					<ToastContent>
						<ToastTitle>Scheduled: Catch up</ToastTitle>
						<ToastDescription>
							Friday, February 10, 2023 at 5:57 PM
						</ToastDescription>
					</ToastContent>
					<ToastProgress />
				</Toast>
			));
		}}>
		Add to calendar
	</Button>
</>
```
